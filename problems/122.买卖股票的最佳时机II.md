```cpp
class Solution
{
public:
    int maxProfit(vector<int> &prices)
    {
        int ans = 0;
        for (int i = 0; i < prices.size(); i++)
        {
            int j = i + 1;
            while (j < prices.size() && prices[j - 1] <= prices[j])
                j++;
            ans += prices[j - 1] - prices[i];
            i = j - 1;
        }
        return ans;
    }
};
```
寻找所有非递减区间，加上高度差即可。
