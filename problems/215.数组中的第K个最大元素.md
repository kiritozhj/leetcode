```cpp
class Solution
{
public:
    int findKthLargest(vector<int> &nums, int k)
    {
        priority_queue<int> pq;
        for (int i : nums)
            pq.push(i);
        int tmp = k - 1;
        while (tmp--)
            pq.pop();
        return pq.top();
    }
};
```
