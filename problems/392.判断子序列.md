```cpp
class Solution
{
public:
    bool isSubsequence(string s, string t)
    {
        int i = 0, j = 0;
        int s_len = s.size(), t_len = t.size();
        while (i < s_len && j < t_len)
        {
            if (s[i] == t[j])
                i++;
            j++;
        }
        if (i == s_len)
            return true;
        return false;
    }
};
```
