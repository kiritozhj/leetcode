class Solution
{
public:
    bool hasCycle(ListNode *head)
    {
        if (!head || !head->next)
        {
            return false;
        }

        ListNode *slow = head;
        ListNode *fast = head;

        // Move slow by 1 step and fast by 2 steps
        while (fast && fast->next)
        {
            slow = slow->next;
            fast = fast->next->next;

            // If slow and fast meet, there's a cycle
            if (slow == fast)
            {
                return true;
            }
        }

        // If fast reaches the end, there's no cycle
        return false;
    }
};
