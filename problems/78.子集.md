class Solution
{
public:
    vector<vector<int>> ans;
    vector<int> cur;

    void backtrack(const vector<int> &nums, int begin)
    {
        ans.push_back(cur);
        if (begin == nums.size())
            return;
        for (int i = begin; i < nums.size(); i++)
        {
            cur.push_back(nums[i]);
            backtrack(nums, i + 1);
            cur.pop_back();
        }
    }

    vector<vector<int>> subsets(vector<int> &nums)
    {
        backtrack(nums, 0);
        return ans;
    }
};
