class Solution
{
public:
    vector<vector<int>> res;
    vector<int> path;

    void dfs(int begin, int n, int k)
    {
        if (path.size() + (n - begin + 1) < k)
            return;
        if (path.size() == k)
        {
            res.push_back(path);
            return;
        }
        path.push_back(begin);
        dfs(begin + 1, n, k);
        path.pop_back();
        dfs(begin + 1, n, k);
    }

    vector<vector<int>> combine(int n, int k)
    {
        dfs(1, n, k);
        return res;
    }
};
